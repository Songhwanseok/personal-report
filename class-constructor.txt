[과제1] 클래스 및 생성자 작성

#include <iostream>
using namespace std;

class CMath
{
private:
 int x, y;
public:
 CMath(int a, int b){ x=a; y=b;}
 int Area(){ return x*y; }
};

int main()
{
 CMath me(4,5);
 cout<<me.Area();
}
